function handleTableChange() { const month = document.getElementById('monthSelect').value; const statusFilter = document.getElementById('statusFilter').value; const searchTerm = document.getElementById('patientSearch').value.toLowerCase(); const tbody = document.getElementById('patientTableBody'); const emptyState = document.getElementById('emptyState'); if (!tbody) return; tbody.innerHTML = ''; let rawList = dbData.rawData || dbData.data || []; // กรองเดือน let filtered = rawList.filter(p => month === 'all' || getVal(p, ['month', 'เดือน']).toString().trim() === month.trim()); // กรองตามสถานะแบบแม่นยำ if (statusFilter !== 'all') { filtered = filtered.filter(p => { const s = getVal(p, ['status', 'สถานะ', 'สถานะชาร์ท']).toString().trim(); const dueStr = getVal(p, ['duestatus', 'สถานะกำหนดส่ง', 'กำหนดส่ง']).toString().trim(); const ldays = parseInt(getVal(p, ['latedays', 'วันค้าง', 'จำนวนวันค้าง']) || 0); // เงื่อนไขที่คุณกำหนด const isOverdue = (dueStr === 'ครบกำหนดส่ง' && s === 'ยังไม่ส่งชาร์ท' && ldays > 1); const isSent = (s === 'ชาร์ทส่งแล้ว'); if (statusFilter === 'เกินกำหนด') return isOverdue; if (statusFilter === 'ส่งแล้ว') return isSent; if (statusFilter === 'รอส่ง') return !isOverdue && !isSent; return true; }); } // ค้นหา Text if (searchTerm) { filtered = filtered.filter(p => { const hn = getVal(p, ['hn', 'รหัสผู้ป่วย']).toString().toLowerCase(); const an = getVal(p, ['an', 'รหัสรับเข้า']).toString().toLowerCase(); const name = getVal(p, ['name', 'patientname', 'ชื่อ', 'ชื่อ-สกุล']).toString().toLowerCase(); return hn.includes(searchTerm) || an.includes(searchTerm) || name.includes(searchTerm); }); } if (filtered.length === 0) { emptyState.style.display = 'block'; return; } emptyState.style.display = 'none'; filtered.forEach(p => { const tr = document.createElement('tr'); const hn = getVal(p, ['hn']); const an = getVal(p, ['an']); const rawName = getVal(p, ['name', 'patientname', 'ชื่อ', 'ชื่อ-นามสกุล']); const adate = getVal(p, ['admitdate', 'วันที่ admit', 'วันที่รับเข้า']); const ddate = getVal(p, ['dcdate', 'วันที่ dc', 'วันที่จำหน่าย']); const dueStr = getVal(p, ['duestatus', 'สถานะกำหนดส่ง', 'กำหนดส่ง']).toString().trim(); const statusStr = getVal(p, ['status', 'สถานะ', 'สถานะชาร์ท']).toString().trim(); const ldays = parseInt(getVal(p, ['latedays', 'วันค้าง', 'จำนวนวันค้าง']) || 0); // คำนวณเพื่อแสดงป้ายสี const isOverdue = (dueStr === 'ครบกำหนดส่ง' && statusStr === 'ยังไม่ส่งชาร์ท' && ldays > 1); const isSent = (statusStr === 'ชาร์ทส่งแล้ว'); let badgeClass = 'bg-warning-light'; let statusText = statusStr !== '-' && statusStr !== '' ? statusStr : 'รอส่ง'; if (isSent) { badgeClass = 'bg-success-light'; statusText = 'ชาร์ทส่งแล้ว'; } else if (isOverdue) { badgeClass = 'bg-danger-light'; statusText = 'ค้างส่ง'; } tr.innerHTML = ` ${hn}${an}${rawName !== "-" ? maskName(rawName) : "-"} ${adate}${ddate} ${dueStr} ${statusText} ${ldays} วัน `; tbody.appendChild(tr); }); }
